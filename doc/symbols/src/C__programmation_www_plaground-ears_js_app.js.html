 34 /**
 35  * Start the App object by using init
 36  * This method create the namespace used all over the application
 37  * @function
 38  * 
 39  * @name PG.App.init
 40  * 
 41  * @param {null} no params
 42  * @return {null} no return
 43  * 
 44  * @this {App}
 45  * 
 46  * @ignore
 47  * 
 48  * @since version 1.0.0
 49  */
 50 App.prototype.init = function ()
 51 {
 52     'use strict';
 53     
 54     // check api key
 55     if(PG.Util.not_null(PG.Settings) && PG.Util.not_null(PG.Settings.apiKey)) {
 56         //PG.User.env = _plgd_settings;
 57         PG.Util.log('APIKEY FOUND = ' + PG.Settings.apiKey);
 58     }else {
 59         PG.Util.log('NO APIKEY FOUND');
 60         return;
 61     }
 62     
 63     PG.App.easyXDM();
 64     PG.App.bindEvent();
 65     PG.User.init();
 66     
 67     PG.Util.fireEvt(
 68         'earsReady',
 69         window,
 70         {
 71             apiKey: PG.Settings.apiKey,
 72             uid: PG.User.uid,
 73             login: PG.User.id
 74 		}
 75 	);
 76 };

 78 /**
 79  * Bind the event to track user move
 80  * @function
 81  * 
 82  * @name PG.App.bindEvent
 83  * 
 84  * @param {null} no params
 85  * @return {null} no return
 86  * 
 87  * @this {App}
 88  * 
 89  * @ignore
 90  * 
 91  * @since version 1.0.0
 92  */
 93 App.prototype.bindEvent = function ()
 94 {
 95     'use strict';
 96     
 97     // AJAX
 98     var xhrListn = new Object();
 99         xhrListn.open = PG.Cache.xmlhttp.prototype.open;
100         xhrListn.send = PG.Cache.xmlhttp.prototype.send;
101     
102     // Catch all the xhr open event
103     PG.Cache.xmlhttp.prototype.open = function (a, b)
104     {
105         if (!a) var a='';
106         if (!b) var b='';
107         
108         xhrListn.open.apply(this, arguments);
109         xhrListn.method = a;  
110         xhrListn.url = b;
111         if (a.toLowerCase() == 'get') {
112             xhrListn.data = b.split('?');
113             xhrListn.data = xhrListn.data[1];
114         }
115     };
116     
117     // Catch all the xhr send event
118     PG.Cache.xmlhttp.prototype.send = function (a, b)
119     {
120         if (!a) var a='';
121         if (!b) var b='';
122         xhrListn.send.apply(this, arguments);
123         if(xhrListn.method.toLowerCase() == 'post') xhrListn.data = a;
124     };
125     
126     // Catch all the load finished event
127     PG.Cache.document.onreadystatechange = function (e)
128     {
129         if(PG.Util.not_null(e.data) && e.data != '') {
130             //PG.App.send(e.data);
131         }
132     };
133     
134     PG.Cache.window.onbeforeunload = function (e)
135     {
136         PG.User.quit();
137         return null;
138     };
139 };

141 /**
142  * Init easyXDM by calling this method
143  * Using easyXDM as RPC mode
144  * @function
145  * 
146  * @name PG.App.easyXDM
147  * 
148  * @param {null} no params
149  * @return {null} no return
150  * 
151  * @this {App}
152  * 
153  * @ignore
154  * 
155  * @since version 1.0.0
156  */
157 App.prototype.easyXDM = function ()
158 {
159     'use strict';
160     
161     PG.App.rpc = new easyXDM.Rpc(
162     {
163         // fallback NameTransport
164         local: PG.Cache.protocol + PG.Config.env[PG.Config.mode].url + "easyXDM/name.html",
165         // fallback swf
166         swf: PG.Cache.protocol + PG.Config.env[PG.Config.mode].url + "easyXDM/easyxdm.swf",
167         // page to load
168         remote: PG.Cache.protocol + PG.Config.env[PG.Config.mode].url + "index.html"
169     },
170     {
171         remote: {
172             request: {}
173         }
174     });
175 };

177 /**
178  * Send a request to easyXDM using RPC with no return
179  * @function
180  * 
181  * @name PG.App.send
182  * 
183  * @param {string} url of the service
184  * @return {Boolean} valid url
185  * 
186  * @this {App}
187  * 
188  * @example
189  * PG.App.send( {String} url)
190  * 
191  * @since version 1.0.0
192  */
193 App.prototype.send = function (url)
194 {
195     'use strict';
196     
197     var broadcast = (PG.User.data.library.config.broadcast),
198         json = {
199             user: {
200                 uid: PG.User.uid
201             },
202             actions: [],
203             page: {
204                 url: url
205             },
206             apiKey: PG.Settings.apiKey
207         },
208         n, o, use, userUrl, story, item, property;
209     
210     if(!PG.Util.isUrlValid(url)) {
211         return false;
212     }
213     
214     // check xpath and url
215     for(n in PG.User.data.library.stories) {
216         story = PG.User.data.library.stories[n];
217         
218         // get variable if send this url
219         if(PG.User.checkStory(story.conditions)) {
220             console.log('push : ')
221             console.log(PG.User.getStory(story.action, story.objects))
222             json.actions.push(PG.User.getStory(story.action, story.objects));
223         }
224     }
225     
226     if(broadcast
227         || json.actions.length > 0) {
228         
229         if(PG.User.isLogged()) {
230             json.user.login = PG.User.id;
231         }
232         json.user = JSON.stringify(json.user);
233         json.actions = JSON.stringify(json.actions);
234         json.objects = JSON.stringify(json.objects);
235         
236         console.log('JSON');
237         console.log(story.user);
238         console.log(story.actions);
239         console.log(story.objects);
240         
241         userUrl = PG.Cache.protocol + PG.Config.env[PG.Config.mode].url + PG.Config.env[PG.Config.mode].send;
242         
243         PG.App.rpc.request(
244             { 
245                 url: userUrl,
246                 method: 'POST',
247                 data: json
248             },
249             function (rpcdata)
250             {
251                 PG.Util.log(rpcdata);
252             },
253             function (error)
254             {
255                 throw 'Api Error, code : ' + error.code + ', msg : ' + error.message;
256             }
257         );
258     }
259     return true;
260 };

262 /**
263  * Send a request to easyXDM using RPC waiting for a json return
264  * @function
265  * 
266  * @name PG.App.call
267  * 
268  * @param {string} serviceName
269  * @return {Object} promise
270  * 
271  * @this {App}
272  * 
273  * @throws {requestError} service cannot be called
274  * 
275  * @example
276  * PG.App.call( {String} service )
277  * .then( {Function} callback)
278  * 
279  * @since version 1.0.0
280  */
281 App.prototype.call = function (s)
282 {
283     'use strict';
284     var promise = new PG.Promise();
285     
286     PG.Util.log('call service : ' + s);
287     
288     PG.App.rpc.request(
289         {
290             url: PG.Cache.protocol + PG.Config.env[PG.Config.mode].url + s + '?apiKey=' + PG.Settings.apiKey
291         },
292         function (rpcdata)
293         {
294             promise.resolve(JSON.parse(rpcdata.data));
295         },
296         function (error)
297         {
298             throw 'Api Error, code : ' + error.code + ', msg : ' + error.message;
299         }
300     );
301     
302     return promise;
303 };
