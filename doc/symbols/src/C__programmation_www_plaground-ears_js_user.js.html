 33     /**
 34      * Init the user object
 35      * @function
 36      * 
 37      * @name PG.User.init
 38      * 
 39      * @param {null}
 40      * @return {null}
 41      * 
 42      * @this {User}
 43      * 
 44      * @ignore
 45      * 
 46      * @since version 1.0.0
 47      */ 
 48     init: function ()
 49     {
 50         'use strict';
 51         
 52         PG.Util.GenerateUniqueId();
 53         PG.User.urls.prev = PG.Util.readCookie('prev-url');
 54         
 55         var authent = PG.Util.readCookie('authent'),
 56             lastSync = PG.Util.readCookie('last-sync'),
 57             today = new Date().getTime();
 58 
 59         if(!PG.Util.not_null(lastSync) || (parseInt(lastSync) + parseInt(24 * 60 * 60)) < today) {
 60             authent = null;
 61         }
 62         
 63         if(!PG.Util.not_null(authent)) {
 64             PG.User.loadAuthent()
 65             .then(function ()
 66             {
 67                 PG.User.checkUser();
 68             });
 69         }else {
 70             PG.User.data = JSON.parse(PG.Util.readCookie('authent'));
 71             PG.User.checkUser();
 72         }
 73     },

 75     /**
 76      * check on cookie if user is logged
 77      * @function
 78      * 
 79      * @name PG.User.isLogged
 80      * 
 81      * @param {null} 
 82      * @return {Boolean} bool true or false
 83      * 
 84      * @this {User}
 85      * 
 86      * @example
 87      * PG.User.isLogged()
 88      * 
 89      * @since version 1.0.0
 90      */
 91     isLogged: function ()
 92     {
 93         'use strict';
 94         
 95         var l;
 96         if(PG.User.id === '' || PG.User.id === null) {
 97             // check
 98             PG.User.id = PG.Util.readCookie('login');
 99         }
100         l = (PG.Util.not_null(PG.User.id)) ? true : false;
101         PG.Util.log('isLogged : ' + l);
102         return l;
103     },
104     
105     /**
106      * Call service to get json data of logout user method
107      * put the return into PG.User.data
108      * @function
109      * 
110      * @name PG.User.loadAuthent
111      * 
112      * @param {null}
113      * @return {Object} promise
114      * 
115      * @this {User}
116      * 
117      * @ignore
118      * 
119      * @since version 1.0.0
120      */
121     loadAuthent: function ()
122     {
123         'use strict';
124         
125         var p = new PG.Promise();
126 
127         PG.App.call( PG.Config.env[PG.Config.mode].connect )
128         .then(
129             function (data)
130             {
131                 PG.User.data = data;
132                 PG.Util.createCookie('last-sync', new Date().getTime());
133                 PG.Util.createCookie('authent', JSON.stringify(data));
134                 p.resolve();
135             }
136         );
137         
138         return p;
139     },

141     /**
142      * Check if the user is logged, logout, or tried to login/logout 
143      * @function
144      * 
145      * @name PG.User.checkUser
146      * 
147      * @param {null}
148      * @return {null}
149      * 
150      * @this {User}
151      * 
152      * @ignore
153      * 
154      * @since version 1.0.0
155      */
156     checkUser: function ()
157     {
158         'use strict';
159         
160         if(PG.Util.not_null(PG.Util.readCookie('logout-try'))) { // USER TRIED TO LOUGOUT
161             if(PG.Util.not_null(PG.User.data.library.stories.logout_user)
162                 && PG.Util.not_null(PG.User.data.library.stories.logout_user.events.after)
163                     && PG.User.checkStory(PG.User.data.library.stories.logout_user.events.after)) {
164                 PG.User.logout();
165             }
166         }else if(PG.Util.not_null(PG.Util.readCookie('login-try'))) { // USER TRIED TO LOGIN
167             if(PG.Util.not_null(PG.User.data.library.stories.login_user)
168                 && PG.Util.not_null(PG.User.data.library.stories.login_user.events.after)
169                     && PG.User.checkStory(PG.User.data.library.stories.login_user.events.after)) {
170                 PG.User.login(PG.Util.readCookie('login-try'));
171             }
172         }
173         if(PG.Util.not_null(PG.Util.readCookie('login'))) { // USER IS ALREADY LOGGED
174             PG.User.login(PG.Util.readCookie('login'));
175         }else { // USER IS LOGGED OUT
176             PG.User.logout();
177         }
178     },

180     /**
181      * Set the login for current user if param {String} not null
182      * call loadLogout to track logout since the user is logged
183      * then return the login
184      * @function
185      * 
186      * @name PG.User.login
187      * 
188      * @param {String} str (optional) if not set just return the login
189      * @return {String} PG.User.data.id
190      * 
191      * @this {User}
192      * 
193      * @example
194      * PG.User.login( {String} username )
195      * 
196      * @since version 1.0.0
197      */
198     login: function (str)
199     {
200         'use strict';
201         
202         PG.Util.eraseCookie('logout-try');
203         PG.Util.log('Login "' + str + '"');
204         if(PG.Util.not_null(str)) {
205             PG.Util.eraseCookie('login-try');
206             PG.User.id = str;
207             PG.Util.createCookie('login', str);
208             PG.App.send(top.location.href);
209         }
210         
211         return PG.User.data.id;
212     },

214     /**
215      * force logout user, then call loadLogin since user is logged out
216      * @function
217      * 
218      * @name PG.User.logout
219      * 
220      * @param {null} 
221      * @return {null}
222      * 
223      * @this {User}
224      * 
225      * @example
226      * PG.User.logout()
227      * 
228      * @since version 1.0.0
229      */
230     logout: function ()
231     {
232         'use strict';
233         
234         PG.Util.log('User is out');
235         PG.Util.eraseCookie('logout-try');
236         PG.Util.eraseCookie('login-try');
237         PG.Util.eraseCookie('login');
238         PG.User.id = null;
239         PG.App.send(top.location.href);
240     },

242     /**
243      * Check evidences for current user.evidences object
244      * @function
245      * 
246      * @name PG.User.checkEvidences
247      * 
248      * @param {null}
249      * @return {Boolean} result true | false
250      * 
251      * @this {User}
252      * 
253      * @ignore
254      * 
255      * @since version 1.0.0
256      */
257     checkEvidences: function (evidences)
258     {
259         'use strict';
260         
261         var obj = null,
262             evidencesOk = true,
263             it;
264         
265         for(it in evidences) {
266             if(evidences[it].selector === 'id') {
267                 obj = document.getElementById(evidences[it].name);
268             }else if(PG.User.data.evidences[it].selector === 'class') {
269                 obj = document.getElementsByClassName(evidences[it].name);
270             }
271             if(obj === null) {
272                 evidencesOk = false;
273             }
274         }
275         
276         return evidencesOk;
277     },

279     /**
280      * Check evidences for current user.evidences object
281      * @function
282      * 
283      * @name PG.User.checkStory
284      * 
285      * @param {Object} story
286      * @return {Boolean} result true | false
287      * 
288      * @this {User}
289      * 
290      * @ignore
291      * 
292      * @since version 1.0.0
293      */
294     checkStory: function (story)
295     {
296         'use strict';
297         
298         var use = false;
299         if(PG.Util.not_null(story)) {
300             // check conditions
301             if(PG.Util.not_null(story)) {
302                 // check condition url + xpath
303                 if(PG.Util.not_null(story.url)
304                     && PG.Util.not_null(story.xpath)
305                         && PG.Util.matchUrl(story.url)
306                             && PG.Util.checkXpath(story.xpath)) {
307                     use = true;
308                 }else if(PG.Util.not_null(story.url)
309                     && PG.Util.matchUrl(story.url)
310                         && !PG.Util.not_null(story.xpath)) {
311                 // check condition url
312                     use = true;
313                 }else if(PG.Util.not_null(story.xpath)
314                     && PG.Util.checkXpath(story.xpath)
315                         && !PG.Util.not_null(story.url)) {
316                 // check condition xpath
317                     use = true;
318                 }
319             }
320         }
321         console.log(story);
322         console.log('use : ' + use);
323         return use;
324     },

326     /**
327      * return json story
328      * @function
329      * 
330      * @name PG.User.getStory
331      * 
332      * @param {String} action
333      * @param {Object} objects
334      * @return {Boolean} result true | false
335      * 
336      * @this {User}
337      * 
338      * @ignore
339      * 
340      * @since version 1.0.0
341      */
342     getStory: function (action, objects)
343     {
344         'use strict';
345         
346         var item = {
347             name: action,
348             properties: []
349         },
350         property, o;
351         
352         if(PG.Util.not_null(objects)) {
353             for(o in objects.properties) {
354                 property = objects.properties[o];
355                 // check condition xpath
356                 if(PG.Util.not_null(property)
357                     && typeof property !== 'function') {
358                     if(PG.Util.not_null(PG.Util.getObjectFromXpath(property.xpath)[0])) {
359                         item.properties.push(
360                             {
361                                 name: property.name,
362                                 value: PG.Util.getValueFromObject(
363                                     PG.Util.getObjectFromXpath(property.xpath)[0]
364                                 )
365                             }
366                         )
367                     }
368                 }
369             }
370         }
371         
372         return item;
373     },

375     /**
376      * Call this method when user quit the current to check if try to logout/login
377      * make cookies 'login-try' and 'logout-try'
378      * @function
379      * 
380      * @name PG.User.quit
381      * 
382      * @param {null}
383      * @return {null}
384      * 
385      * @this {User}
386      * 
387      * @ignore
388      * 
389      * @since version 1.0.0
390      */
391     quit: function ()
392     {
393         'use strict';
394         
395         var it,
396             id = '',
397             s;
398         
399         PG.Util.createCookie('prev-url', top.location.href);
400         
401         if(PG.User.isLogged()) {
402             // test logout
403             if(PG.Util.not_null(PG.User.data.library.stories.logout_user)
404                 && PG.Util.not_null(PG.User.data.library.stories.logout_user.events.before)) {
405                 if(PG.User.checkStory(PG.User.data.library.stories.logout_user.events.before)) {
406                     s = PG.User.getStory(
407                         PG.User.data.library.stories.logout_user.action,
408                         PG.User.data.library.stories.logout_user.objects
409                     );
410                     PG.Util.createCookie('logout-try', JSON.stringify(s));
411                 }
412             }
413         }else {
414             // test login
415             if(PG.Util.not_null(PG.User.data.library.stories.login_user)
416                 && PG.Util.not_null(PG.User.data.library.stories.login_user.events.before)) {
417                 if(PG.User.checkStory(PG.User.data.library.stories.login_user.events.before)) {
418                     s = PG.User.getStory(
419                         PG.User.data.library.stories.login_user.action,
420                         PG.User.data.library.stories.login_user.objects
421                     );
422                     PG.Util.createCookie('login-try', JSON.stringify(s));
423                 }
424             }
425         }
426         
427         return;
428 
429         if(PG.Util.not_null(PG.User.data.login) && PG.Util.not_null(PG.User.data.logout)) {
430             // check user is on login page 
431             if(!PG.User.isLogged() && PG.Util.matchUrl(PG.User.data.login.urls.page)) {
432                 for(it in PG.User.data.login.items) {
433                     if(typeof PG.User.data.login.items[it] === 'object') {
434                         id += (id !== '') ? ':' : '';
435                         id += PG.Util.getValueFromObject(
436                             PG.Util.getDomElemntFromItem(PG.User.data.login.items[it])
437                         );
438                     }
439                 }
440                 PG.Util.createCookie('login-try', id);
441             // check user is on logout page
442             }else if(PG.Util.matchUrl(PG.User.data.logout.urls.page)) {
443                 PG.Util.createCookie('logout-try', 'logout');
444             }
445         }
446     }
