 25 pl.ready(function ()
 26 {
 27     'use strict';
 28     
 29     /**
 30      * Implémentation du pattern promise
 31      * @constructor
 32      * @param {null} no params
 33      * @return {null} no return
 34      * @this {Object} promise
 35      */
 36     var promise = function ()
 37     {
 38         'use strict';
 39     };
 40     
 41     promise.prototype = {
 42         pending:null,
 43         
 44         /**
 45          * function called by other with 2 method
 46          * @param {Function} success callback
 47          * @param {Function} failure callback
 48          * @return {object} this
 49          * @this {object} promise
 50          */
 51         then: function (success, failure)
 52         {
 53             'use strict';
 54             
 55             this.pending = { resolve: success, reject: failure };
 56             return this;
 57         },
 58         
 59         /**
 60          * never call this method, use then (success, failure)
 61          * @param {object} value
 62          * @return {null} no return
 63          * @this {object} promise
 64          */
 65         resolve: function (value)
 66         {
 67             'use strict';
 68             
 69             this.pending.resolve(value);
 70         },
 71         
 72         /**
 73          * never call this method, use then (success, failure)
 74          * @param {object} value
 75          * @return {null} no return
 76          * @this {object} promise
 77          */
 78         reject: function (error)
 79         {
 80             'use strict';
 81             
 82             this.pending.reject(error);
 83         }
 84     };
 85     
 86     // put the promise into Adfab.Playground.Promise
 87     try {
 88         addToNamespace('Promise', promise);
 89     }catch(e) {
 90        throw new Error( "Cannot extends 'Promise' to 'Adfab.playground.Promise'" );
 91     }
 92 });
