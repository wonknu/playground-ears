 18 pl.ready(function ()
 19 {
 20 	/**
 21 	 * Cache Object
 22 	 * @class
 23      * @name PG.Cache
 24 	 */
 25     cache = {
 26 		/**
 27 		 * @name  PG.Cache.window
 28 		 * @constant PG.Cache.window
 29 		 * @type Object
 30 		 */
 31         window: window,
 32 		/**
 33 		 * @name  PG.Cache.document
 34 		 * @constant PG.Cache.document 
 35 		 * @type Object
 36 		 */
 37         document: document,
 38 		/**
 39 		 * @name  PG.Cache.xmlhttp
 40 		 * @constant PG.Cache.xmlhttp 
 41 		 * @type Object
 42 		 */
 43         xmlhttp: window.XMLHttpRequest,
 44 		/** Protocol used in current page (http | https)
 45 		 * @name  PG.Cache.protocol
 46 		 * @constant PG.Cache.protocol 
 47 		 * @type String
 48 		 */
 49         protocol: ('https:' == document.location.protocol ? 'https://' : 'http://'),
 50 		/** isTouchDevice tell if the current user is a mobile device
 51 		 * @name  PG.Cache.isTouchDevice
 52 		 * @constant PG.Cache.isTouchDevice
 53 		 * @type boolean
 54 		 */
 55         isTouchDevice: (!!('ontouchstart' in window) || !!('onmsgesturechange' in window)),
 56 		/** clickEvent used in the application can be touch event if mobile user
 57 		 * @name  PG.Cache.clickEvent
 58 		 * @constant PG.Cache.clickEvent 
 59 		 * @type Object
 60 		 */
 61         clickEvent: (!!('ontouchstart' in window) || !!('onmsgesturechange' in window)) ? 'touchstart' : 'click',
 62 		/** config create from "pl_config" variable
 63 		 * @name  PG.Cache.config
 64 		 * @constant PG.Cache.config 
 65 		 * @type Object
 66 		 */
 67         config: pl_config,
 68 		/** config create from "_plgd_settings" variable
 69 		 * @name  PG.Cache.settings
 70 		 * @constant PG.Cache.settings 
 71 		 * @type Object
 72 		 */
 73         settings: _plgd_settings,
 74 		/** check if localstorage is available in the current device
 75 		 * @name  PG.Cache.localstorage
 76 		 * @constant PG.Cache.localstorage 
 77 		 * @type boolean
 78 		 */
 79         localstorage: ((typeof window.localStorage == 'undefined' || typeof window.sessionStorage == 'undefined') ? false : true)
 80     };
 81     
 82     // put the cache into Adfab.Playground.Cache
 83     try {
 84         addToNamespace('Cache', cache);
 85     }catch(e) {
 86        throw new Error( "Cannot extends 'cache' to 'Adfab.playground.Cache'" );
 87     }
 88 });
