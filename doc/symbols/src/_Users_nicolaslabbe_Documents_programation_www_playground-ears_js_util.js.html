 33         /** 
 34          * Check if a variable is null
 35 	     * @function 
 36          * 
 37 	     * @name PG.Util.not_null
 38 	     * 
 39          * @param {Object} object DOM object
 40          * @return {boolean} bool true or false  
 41 	     * 
 42 	     * @this {Util}
 43 	     * 
 44 	     * @example
 45 	     * PG.Util.not_null( Object | String | ... )
 46 	     * 
 47 	     * @since version 1.0.0
 48          */
 49         not_null: function (obj)
 50         {
 51             'use strict';
 52             
 53             return ((obj != null) ? ((typeof(obj) != 'undefined') ? ((obj !== '') ? true : false) : false) : false);
 54         },

 56         /** 
 57          * Return true if xpath is found
 58 	     * @function 
 59          * 
 60 	     * @name PG.Util.checkXpath
 61 	     * 
 62          * @param {String} xpath 
 63          * @return {boolean} bool true or false  
 64 	     * 
 65 	     * @this {Util}
 66 	     * 
 67 	     * @example
 68 	     * PG.Util.checkXpath( {String} xpath )
 69 	     * 
 70 	     * @since version 1.0.0
 71          */
 72         checkXpath: function (xpath)
 73         {
 74         	var result = document.evaluate(
 75 	        		xpath,
 76 	        		document.documentElement,
 77 	        		null,
 78 	        		XPathResult.ORDERED_NODE_ITERATOR_TYPE,
 79 					null
 80 				),
 81 				cpt = 0;
 82 				
 83 				if (result) {
 84 				    var node = result.iterateNext();
 85 				    while(node) {
 86 				        cpt++;
 87 				        node = result.iterateNext();
 88 				    }
 89 				}
 90 				return (cpt > 0) ? true : false;
 91         },

 93         /**
 94          * Return the content of the given tag
 95 	     * @function
 96          * 
 97 	     * @name PG.Util.getValueFromObject
 98 	     * 
 99          * @param {Object} DOM Object
100          * @return {String} value
101 	     * 
102 	     * @this {Util}
103 	     * 
104 	     * @example
105 	     * PG.Util.getValueFromObject( Object | String | ... )
106 	     * 
107 	     * @since version 1.0.0
108          */
109         getValueFromObject: function (obj)
110         {
111             'use strict';
112             
113             var value = '';
114             if(PG.Util.not_null(obj) && typeof obj == 'object' && obj.type != undefined) {
115                 if(obj.value != undefined) {
116                     value = obj.value;
117                 }else if(obj.innerHTML != undefined) {
118                     value = obj.innerHTML;
119                 }
120             }
121             return value;
122         },

124         /**
125          * Create a cookie under Adfab.Playground.[ name ]
126 	     * @function
127 	     * 
128 	     * @name PG.Util.createCookie
129 	     * 
130          * @param {String} name
131          * @param {String} value
132          * @param {Number} [number] of days (optional)
133          * @return {null} no return
134          * 
135 	     * @this {Util}
136 	     * 
137 	     * @example
138 	     * PG.Util.createCookie( {String} name, {String} value, [ {Number} days ])
139 	     * 
140 	     * @since version 1.0.0
141          */
142         createCookie: function (name, value, days)
143         {
144             'use strict';
145             
146             // if localstorage enabled
147             if(PG.Cache.localstorage) {
148             	localStorage.setItem(PG.Cache.config.ns + '.' + name, value);
149         	// if localstorage not enable use cookies as polyfill
150             }else {
151 	            if (days) {
152 	                var date = new Date();
153 	                date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
154 	                var expires = "; expires=" + date.toGMTString();
155 	            }
156 	            else var expires = "";
157 	            document.cookie = PG.Cache.config.ns + '.' + name + "=" + encodeURIComponent(value) + expires + "; path=/";
158             }
159         },

161         /**
162          * Get value of a cookie
163 	     * @function
164 	     * 
165 	     * @name PG.Util.readCookie
166 	     * 
167          * @param {String} name
168          * @return {String} | {null} value of the cookie
169          * 
170 	     * @this {Util}
171 	     * 
172 	     * @example
173 	     * PG.Util.readCookie( {String} name)
174 	     * 
175 	     * @since version 1.0.0
176          */
177         readCookie: function (name)
178         {
179             'use strict';
180             
181             // if localstorage enabled
182             if(PG.Cache.localstorage) {
183             	return localStorage.getItem(PG.Cache.config.ns + '.' + name);
184         	// if localstorage not enable use cookies as polyfill
185             }else {
186 	            var nameEQ = PG.Cache.config.ns + '.' + name + "=",
187 	                ca = document.cookie.split(';'),
188 	                i = 0,
189 	                c = '';
190 	                
191 	            for(i = 0; i < ca.length; i++) {
192 	                c = ca[i];
193 	                while (c.charAt(0)==' ') c = c.substring(1,c.length);
194 	                    if (c.indexOf(nameEQ) == 0) return decodeURIComponent(c.substring(nameEQ.length, c.length));
195 	            }
196 			}
197             return null;
198         },

227 			}

230         /**
231          * Tell if input URL match current URL
232 	     * @function
233 	     * 
234 	     * @name PG.Util.matchUrl
235 	     * 
236          * @param {String} url
237          * @return {Boolean} bool true or false
238          * 
239 	     * @this {Util}
240 	     * 
241 	     * @example
242 	     * PG.Util.matchUrl( {String} url)
243 	     * 
244 	     * @since version 1.0.0
245          */
246         matchUrl: function (url)
247         {
248             'use strict';
249             try {
250 	            url = eval(url);
251 	            return (url.test(top.location.href));
252             }catch (e) {
253             	return false;
254             }
255         },

302        /**
303         * get dom element from object.selector and object.name
304 	    * @function
305 	    * 
306 	    * @name PG.Util.getDomElemntFromItem
307 	    * 
308         * @param {Object} item object with variable
309         * @param item.selector type of the DOM element
310         * @param item.name id or class of the DOM element
311         * @return {Object} obj DOM object
312         * 
313 	    * @this {Util}
314 	    * 
315 	    * @example
316 	    * PG.Util.getDomElemntFromItem( {String} it)
317 	    * 
318 	    * @since version 1.0.0
319         */
320        getDomElemntFromItem: function (it)
321        {
322             var obj = null,
323                 o;
324            
325             if(it.selector === 'id') {
326                 obj = document.getElementById(it.name);
327             }else if(it.selector === 'class') {
328                 obj = document.getElementsByClassName(it.name);
329                 if(obj !== null && obj.length > 0) {
330                     obj = obj[0];
331                 }
332             }
333            
334             return obj;
335 		},

337        /**
338         * polyfill function to trigger custom event
339 	    * @function
340 	    * 
341 	    * @name PG.Util.fireEvtd
342 	    * 
343         * @param {String} name of the event
344         * @param {Object} dom element to trigger the event (use window)
345         * @param {Object} data send to the event handler
346         * @return {null}
347         * 
348 	    * @this {Util}
349 	    * 
350 	    * @example
351 	    * PG.Util.fireEvt( {String} name, {Object} window, {Object} data)
352 	    * 
353 	    * @since version 1.0.0
354         */
355 		fireEvt: function (type, obj, data)
356 		{ 
357 			if(window.dispatchEvent) {
358 				var event = document.createEvent('Event');
359 				event.initEvent(type, true, true);
360 				event.data = data;
361 				obj.dispatchEvent(event);
362 			}else if(window.fireEvent) {
363 	        	obj.fireEvent(type);
364 			}
365 		}
